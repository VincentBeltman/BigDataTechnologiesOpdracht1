-- Function: new_album(character varying, date, character varying, integer)

-- DROP FUNCTION new_album(character varying, date, character varying, integer);

CREATE OR REPLACE FUNCTION new_album(
    name character varying,
    release_date date,
    publisher character varying,
    artist integer)
  RETURNS void AS
$BODY$BEGIN
INSERT INTO album(
          name, release_date, publisher_organisation, artist_id)
    VALUES (name, release_date, publisher, artist);
End$BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;
ALTER FUNCTION new_album(character varying, date, character varying, integer)
  OWNER TO postgres;
